def text_to_tensor(self, text: Text):
    # Convert text into fixed-size tensor
    text = text.strip().lower()
    max_length = 100
    vector = [ord(char) % 100 for char in text[:max_length]]  # ASCII-based encoding
    vector += [0] * (max_length - len(vector))  # Padding
    return vector
