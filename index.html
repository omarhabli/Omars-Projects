import torch
import torch.nn as nn
import torch.optim as optim
from typing import Any, Dict, List, Text, Optional

class SelfAttention(nn.Module):
    def __init__(self, input_dim: int):
        super(SelfAttention, self).__init__()
        self.query = nn.Linear(input_dim, input_dim)
        self.key = nn.Linear(input_dim, input_dim)
        self.value = nn.Linear(input_dim, input_dim)
        self.softmax = nn.Softmax(dim=-1)

    def forward(self, x):
        Q = self.query(x)
        K = self.key(x)
        V = self.value(x)
        attention_scores = self.softmax(torch.matmul(Q, K.transpose(-2, -1)))
        return torch.matmul(attention_scores, V)

class AdvancedIntentClassifierModel(nn.Module):
    def __init__(self, input_dim: int, num_intents: int):
        super(AdvancedIntentClassifierModel, self).__init__()
        self.layers = nn.Sequential(
            nn.Linear(input_dim, 128), 
            nn.ReLU(),
            nn.Dropout(0.3),
            SelfAttention(128), 
            nn.Linear(128, 64), 
            nn.BatchNorm1d(64),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(64, num_intents)
        )

    def forward(self, x):
        return self.layers(x)

def text_to_tensor(text: Text, max_length: int = 100):
    text = text.strip().lower()
    vector = [ord(char) % 100 for char in text[:max_length]]
    vector += [0] * (max_length - len(vector))
    return vector
            
class AdvancedPytorchIntentClassifier:
    def __init__(self, num_intents: int):
        self.num_intents = num_intents
        self.model = AdvancedIntentClassifierModel(100, num_intents)  # Input size: 100
        self.optimizer = optim.Adam(self.model.parameters(), lr=0.001)
        self.criterion = nn.CrossEntropyLoss()
        self.scheduler = optim.lr_scheduler.CosineAnnealingLR(self.optimizer, T_max=20, eta_min=0.0001)
        self.intents = []
        self.model_path = "advanced_model.pth"

    def train(self, input_texts: List[torch.Tensor], labels: List[torch.Tensor], epochs: int = 50):
        self.model.train()
        for epoch in range(epochs):
            running_loss = 0.0
            for text_tensor, label_tensor in zip(input_texts, labels):
                self.optimizer.zero_grad()
                outputs = self.model(text_tensor.unsqueeze(0).float())
                loss = self.criterion(outputs, label_tensor.unsqueeze(0))
                loss.backward()
                self.optimizer.step()
                running_loss += loss.item()

            self.scheduler.step()
            print(f"Epoch [{epoch + 1}/{epochs}], Loss: {running_loss / len(input_texts)}")

        torch.save(
            {"model_state_dict": self.model.state_dict(), "intents": self.intents},
            self.model_path,
        )

    def prepare_training_data(self, training_data: List[Dict[Text, Any]]):
        input_texts = []
        labels = []
        for example in training_data:
            text = example.get("text")
            intent = example.get("intent")
            if text and intent:
                text_tensor = torch.tensor(text_to_tensor(text), dtype=torch.float32)
                label_tensor = torch.tensor(self.intents.index(intent), dtype=torch.long)
                input_texts.append(text_tensor)
                labels.append(label_tensor)
        return input_texts, labels

    def load_model(self):
        if torch.exists(self.model_path):
            checkpoint = torch.load(self.model_path)
            self.model.load_state_dict(checkpoint["model_state_dict"])
            self.intents = checkpoint["intents"]
            self.model.eval()

            
class AdvancedPytorchIntentClassifier:
    # Ensure initialization and loading logic remain here as well.
    def __init__(self, num_intents: int):
        self.num_intents = num_intents
        self.model = AdvancedIntentClassifierModel(100, num_intents)
        self.intents = []
        self.model_path = "advanced_model.pth"

    def process(self, messages: List[Dict[Text, Any]]) -> List[Dict[Text, Any]]:
        if self.model is None:
            self.load_model()

        results = []
        for message in messages:
            text = message.get("text")
            if not text:
                continue

            text_tensor = torch.tensor([text_to_tensor(text)], dtype=torch.float32)
            with torch.no_grad():
                predictions = self.model(text_tensor)
                predicted_label_idx = predictions.argmax(dim=1).item()
                intent_name = self.intents[predicted_label_idx]
                confidence = torch.softmax(predictions, dim=1).max().item()
                results.append({"text": text, "intent": intent_name, "confidence": confidence})

        return results

    def load_model(self):
        if torch.exists(self.model_path):
            checkpoint = torch.load(self.model_path)
            self.model.load_state_dict(checkpoint["model_state_dict"])
            self.intents = checkpoint["intents"]
            self.model.eval()
